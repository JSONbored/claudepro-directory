// Create Zod schemas from Prisma enum object VALUES (these match database values)
// These are needed because Prisma enum objects have keys (snake_case) that map to database values (kebab-case)
// Use z.nativeEnum() to properly infer enum types instead of literal union types

export const jobCategorySchema: z.ZodType<job_category> = z.nativeEnum(JobCategory);

// Create properly typed enum schemas using z.nativeEnum() with enum objects
