#!/bin/bash

# Docker Health Statusline
# Real-time container health monitoring for Claude Code CLI

# Get running containers count
running=$(docker ps -q 2>/dev/null | wc -l | tr -d ' ')

# Get total containers
total=$(docker ps -a -q 2>/dev/null | wc -l | tr -d ' ')

# Get unhealthy containers
unhealthy=$(docker ps --filter health=unhealthy -q 2>/dev/null | wc -l | tr -d ' ')

# Color codes
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
RED='\033[0;31m'
BLUE='\033[0;34m'
NC='\033[0m' # No Color

# Status indicator
if [ "$unhealthy" -gt 0 ]; then
    status="${RED}â—${NC}"
elif [ "$running" -eq 0 ]; then
    status="${YELLOW}â—‹${NC}"
else
    status="${GREEN}â—${NC}"
fi

# Display statusline
echo -e "${BLUE}ğŸ³${NC} ${status} ${running}/${total}"
