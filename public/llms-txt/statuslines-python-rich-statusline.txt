#!/usr/bin/env python3

import sys
import json
from rich.console import Console
from rich.text import Text

console = Console()

# Read JSON from stdin
try:
    data = json.load(sys.stdin)
except json.JSONDecodeError:
    console.print("[red]Error: Invalid JSON input[/red]")
    sys.exit(1)
